\begin{frame}{Motivation \& Objective}
  \begin{block}{Objective}
    To optimize the Density Functional Theory (DFT) electron density prediction workflow by eliminating expensive Self-Consistent Field (SCF) calculations.
  \end{block}
  
  \begin{itemize}
      \item \textbf{Current Bottleneck:} SCF cycles in GPAW are computationally expensive.
      \item \textbf{Proposed Solution:} Replace post-SCF input ($v_{ext}$) with a pre-SCF ionic potential ($v_{ion}$) obtained via \texttt{maxiter=0}.
      \item \textbf{Model:} Train an SO(3)-Equivariant 3D U-Net to map $v_{ion} \rightarrow n_r$ (ground-state electron density).
      \item \textbf{Why Equivariant?} Electron density must rotate identically with the input molecule geometry. E(3) / SO(3) architectures guarantee physical symmetry-awareness without massive data augmentation.
  \end{itemize}
\end{frame}

% SLIDE 3: DATASET & ARCHITECTURE

\begin{frame}{Dataset \& Model Architecture}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{QM9 Phase A Dataset}
      \begin{itemize}
          \item Smallest 1,000 molecules from QM9.
          \item Input: $v_{ion}$ (effective potential at $n_0$, shape $X \times Y \times Z$).
          \item Target: $n_r$ (full SCF pseudo-density).
          \item DFT Settings: PAW setups, LDA functional (to be upgraded to PBE/HGH).
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{SO(3)-Equivariant 3D U-Net}
      \begin{itemize}
          \item Built with \texttt{escnn.nn.R3Conv}.
          \item Parameters: $\sim$542,000.
          \item \texttt{base\_channels} = 32.
          \item \texttt{max\_freq} = 2 (SO(3) frequency).
          \item \textbf{Crucial Fix:} Replaced \texttt{ReLU} with \texttt{Softplus} at the output and added \textit{per-sample input standardization} to prevent Dead Neurons (Dying ReLU problem) caused by the large scale of $v_{ion}$ (-127 eV to 0 eV).
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Equivariance Validation (Corrected)}
  \begin{block}{Methodology}
    We verified model equivariance under \textbf{nearest-neighbor} (order=0, no interpolation) rotations: \\
    \texttt{scipy.ndimage.rotate(..., order=0)} — pure voxel permutation, no smoothing.
  \end{block}

  \begin{table}[h]
    \centering
    \scriptsize
    \caption{R$^2$ under 90$^\circ$ and 180$^\circ$ rotation (order=0, 10 molecules).}
    \begin{tabular}{lcc}
    \toprule
    Metric & 90$^\circ$ Rotation & 180$^\circ$ Rotation \\
    \midrule
    Best R$^2$ & 0.9941 (gdb\_130574, C$_3$N$_4$O$_2$) & 0.9941 (gdb\_130574) \\
    Worst R$^2$ & \textbf{0.7745} (gdb\_3820, C$_3$HN$_3$O) & 0.9866 (gdb\_130) \\
    Mean R$^2$ & 0.9344 & \textbf{0.9893} \\
    Mean MAE & 0.0165 & 0.0073 \\
    \bottomrule
    \end{tabular}
  \end{table}

  \begin{itemize}
      \item \textbf{180$^\circ$ Rotations (robust):} Pure sign-inversion. R$^2 > 0.986$ for all molecules. Model handles reflection symmetries perfectly.
      \item \textbf{90$^\circ$ Rotations (grid-dependent):} Axis swapping causes significant degradation on \textbf{anisotropic grids} (e.g., $32\times48\times20 \rightarrow 20\times48\times32$). The scalar-only \texttt{escnn} representation ($\ell=0$, \texttt{trivial\_repr}) does \textbf{not} guarantee equivariance on arbitrary non-cubic 3D grids.
  \end{itemize}

  \begin{alertblock}{Important Distinction}
    GPAW itself is rigorously equivariant ($\Delta E < 10^{-10}$~eV for 90$^\circ$/180$^\circ$). The degradation observed here is a \textbf{model limitation}, not a DFT/PAW limitation. See next slides for PAW injection validation.
  \end{alertblock}
\end{frame}

% SLIDE 8a: PAW CONSISTENCY UNDER ROTATION
% -----------------------------------------------------------------------------
\begin{frame}{PAW Consistency Under Rotation: Why $D_{asp}$ Must Rotate}
  \begin{block}{The PAW Decomposition}
    $n(\mathbf{r}) = \tilde{n}(\mathbf{r}) + \sum_a \big[ n_a(\mathbf{r}) - \tilde{n}_a(\mathbf{r}) \big]$
    \begin{itemize}
      \item $\tilde{n}$: smooth pseudo-density on the grid.
      \item $n_a - \tilde{n}_a$: augmentation-sphere correction reconstructed from $D_{asp}$.
    \end{itemize}
  \end{block}

  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{If only $\tilde{n}$ is rotated:}
      \begin{itemize}
        \item Grid says bond points along $+y$.
        \item $D_{asp}$ still describes old $+x$ angular distribution.
        \item \textcolor{red}{Multipole moments (dipole, quadrupole) are wrong.}
        \item \textcolor{red}{Hartree + XC energy corrupted $\rightarrow$ error $\sim$1000~eV.}
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Solution: Rotate $D_{asp}$ with Wigner-D matrices}
      \begin{itemize}
        \item GPAW provides \texttt{gpaw.rotation.rotation(l, $U_{vv}$)}.
        \item Build block-diagonal $R_{ii}$ per atom.
        \item Transform: $D'_{ii} = R_{ii} \, D_{ii} \, R_{ii}^\top$.
      \end{itemize}
      \vspace{0.2cm}
      \textbf{Convention trap:}
      \begin{itemize}
        \scriptsize
        \item ASE: column-vector ($\mathbf{r}' = R\mathbf{r}$).
        \item GPAW: row-vector ($\mathbf{r}'^\top = \mathbf{r}^\top U$), so $U = R^\top$.
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.3cm}
  \centering
  \footnotesize
  \textbf{Result:} Full SCF on 90$^\circ$/180$^\circ$ yields $E$ identical to original ($\Delta E < 10^{-10}$~eV). GPAW itself is perfectly equivariant.
\end{frame}

% SLIDE 8b: THREE CRITICAL BUGS & FIXES
% -----------------------------------------------------------------------------
\begin{frame}{Lessons Learned: Three Critical Implementation Bugs}
  \begin{enumerate}
    \item \textbf{Padding Bug}
      \begin{itemize}
        \item \texttt{get\_pseudo\_density(pad=True)} adds a zero-layer at index~0.
        \item After \texttt{scipy.ndimage.rotate}, padding is no longer at the boundary.
        \item \textbf{Fix:} Extract unpadded density directly from \texttt{calc.density.nt\_sG[0]} (e/Bohr$^3$).
      \end{itemize}

    \vspace{0.2cm}
    \item \textbf{Handedness Bug}
      \begin{itemize}
        \item \texttt{scipy.ndimage.rotate($+90^\circ$, axes=(0,2))} rotates \textbf{opposite} to \texttt{ase.Atoms.rotate($+90^\circ$, 'y')}.
        \item Empirically: ASE $+90^\circ$~y = scipy $-90^\circ$~axes(0,2).
        \item \textbf{Fix:} Use $-90^\circ$ in scipy to match ASE geometry rotation.
      \end{itemize}

    \vspace{0.2cm}
    \item \textbf{$D_{asp}$ Rotation Bug}
      \begin{itemize}
        \item Without rotating $D_{asp}$, injection energy error = \textbf{$\sim$1000~eV} (not a typo).
        \item The PAW augmentation spheres contain the wrong angular character.
        \item \textbf{Fix:} Apply Wigner-D $R_{ii} D_{ii} R_{ii}^\top$ with row-vector convention $U = R^\top$.
      \end{itemize}
  \end{enumerate}

  \vspace{0.2cm}
  \centering
  \footnotesize
  All three bugs were discovered and fixed during a single session on gdb\_1006 (C$_6$H$_4$O).\\
  Script: \texttt{scripts/test\_gdb1006\_lda\_rotation\_injection.py}
\end{frame}
