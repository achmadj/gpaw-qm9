\begin{frame}{Motivation \& Objective}
  \begin{block}{Objective}
    To optimize the Density Functional Theory (DFT) electron density prediction workflow by eliminating expensive Self-Consistent Field (SCF) calculations.
  \end{block}
  
  \begin{itemize}
      \item \textbf{Current Bottleneck:} SCF cycles in GPAW are computationally expensive.
      \item \textbf{Proposed Solution:} Replace post-SCF input ($v_{ext}$) with a pre-SCF ionic potential ($v_{ion}$) obtained via \texttt{maxiter=0}.
      \item \textbf{Model:} Train an SO(3)-Equivariant 3D U-Net to map $v_{ion} \rightarrow n_r$ (ground-state electron density).
      \item \textbf{Why Equivariant?} Electron density must rotate identically with the input molecule geometry. E(3) / SO(3) architectures guarantee physical symmetry-awareness without massive data augmentation.
  \end{itemize}
\end{frame}

\begin{frame}{Why Dense CNNs Also Handle Rotation}
  \begin{block}{Dense CNNs lack explicit SO(3) equivariance — yet they still work}
    The task ($v_{ion} \rightarrow n_r$) has properties that confer \textit{approximate} rotational robustness.
  \end{block}

  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{1. Isotropic 3$\times$3$\times$3 Kernels}
      \begin{itemize}
        \item Dense CNN kernels are small cubes; locally they do not distinguish x/y/z.
        \item This gives \textbf{approximate rotational invariance} for local features.
      \end{itemize}

      \vspace{0.2cm}
      \textbf{2. Field-to-Field Regression}
      \begin{itemize}
        \item Mapping $v_{ion} \rightarrow n_r$ is dominated by local physics.
        \item ``High potential near nucleus $\rightarrow$ high density'' holds regardless of global orientation.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{3. Training Diversity}
      \begin{itemize}
        \item QM9 contains molecules in random orientations.
        \item Dense model learns \textbf{orientation-averaged} features.
      \end{itemize}

      \vspace{0.2cm}
      \textbf{4. The Critical Difference}
      \begin{itemize}
        \item Dense CNN has \textbf{no mathematical guarantee} of SO(3) equivariance.
        \item On extreme anisotropy, it degrades (C$_5$HNO$_2$: dense 0.989 vs equi 0.998).
        \item Explicit equivariance provides \textbf{provable} generalization.
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.3cm}
  \centering
  \footnotesize
  \textbf{Takeaway:} Dense CNNs work ``by accident'' (local isotropy + training diversity). Equivariant architecture works ``by design."
\end{frame}
